private variable
  m n l : ℕ
  A B C : Set ℓ

pattern tt = lift tt₀

data Mode : Set where
  Chk Syn : Mode

Chk≢Syn : Chk ≢ Syn
Chk≢Syn ()

infixl 4 _^_
_^_ : Set ℓ → ℕ → Set ℓ
X ^ n = Vec X n

Fins : ℕ → Set
Fins = List ∘ Fin

Lift₀ : {ℓ : Level} → Set ℓ → Set ℓ
Lift₀ {ℓ} = Lift {ℓ} 0ℓ

{-# DISPLAY Lift lzero A = Lift₀ A #-}

infixl 6 _ʳ+_
_ʳ+_ : ℕ → ℕ → ℕ
zero  ʳ+ m = m
suc n ʳ+ m = n ʳ+ (suc m)

data And : List Bool → Bool → Set where
  nil :                       And []           true
  hd  : ∀ {bs}              → And (false ∷ bs) false
  tl  : ∀ {bs b} → And bs b → And (true  ∷ bs) b

index-∈-lookup
  : ∀ {ℓ} {A : Set ℓ} (xs : List A) (i : Fin (length xs))
  → index (∈-lookup {xs = xs} i) ≡ i
index-∈-lookup (x ∷ xs) zero    = refl
index-∈-lookup (x ∷ xs) (suc i) = cong suc (index-∈-lookup xs i)
------------------------------------------------------------------------------
-- Properties of _≗_

≗-isEquivalence : IsEquivalence (_≗_ {A = A} {B = B})
≗-isEquivalence = record
  { refl  = λ {f} x → refl
  ; sym   = λ {f} {g} eq → sym ∘ eq
  ; trans = λ {f} {g} {h} f=g g=h x → trans (f=g x) (g=h x)
  }

open import Prelude.DecEq public

instance
  EqNat : DecEq ℕ
  _≟_ ⦃ EqNat ⦄ = N._≟_

  EqFin : DecEq (Fin n)
  _≟_ ⦃ EqFin ⦄ = F._≟_

instance
  NumNat : Number ℕ
  NumNat = N.number

  NumFin : {n : ℕ} → Number (Fin n)
  NumFin {n} = F.number n
